1. charAt/charCodeAt

str = "강아지/라쿤/고양이/사막여우/북극여우/라쿤/미어캣"
charAt(index) : index위치의 문자를 리턴
charCodeAt(index) : index위치의 unicode값을 리턴



2. indexOf/lastIndexOf

str = "강아지/라쿤/고양이/사막여우/북극여우/라쿤/미어캣"
indexOf("문자열") : 문자열의 위치를 검색
lastIndexOf("문자열") : 문자열의 위치를 검색 - 뒤에서부터 시작



3. concat/trim

str = "강아지/라쿤/고양이/사막여우/북극여우/라쿤/미어캣"
str2 = "사과/복숭아/딸기/레몬/바나나/배/수박/토마토"
concat() : 문자열 합치기
str4 = "     사과/복숭아/딸기/레몬/바나나/배/수박/토마토    "
trim() : 문자열의 앞뒤 공백 제거



4. substring/slice/substr

str = "강아지/라쿤/고양이/사막여우/북극여우/라쿤/미어캣"
substring(start, end) : start ~ end-1 까지의 글자를 추출
slice(start, end) : start ~ end-1 까지의 글자를 추출
substr(start, length) : start 부터 length갯수를 추출



5. split

str = "강아지/라쿤/고양이/사막여우/북극여우/라쿤/미어캣"
split(구분자, 갯수) : 구분자로 문자를 구분해서 배열로 반환한다



6. replace

str = "강아지/라쿤/고양이/사막여우/북극여우/라쿤/미어캣"
replace(old, new) : old의 값을 new의 값으로 변경
replace(//g) : /와 /사이에 정규표현식을 사용할 수 있다. g : global
